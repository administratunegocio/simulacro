def ob_usuario(usuario):
	cone = obtener_bd()
	cursor = cone.cursor(dictionary=True)

	query = "SELECT id_user, user, password FROM usuarios WHERE user = %s;"
	values = (usuario,)

	try:
		cursor.execute(query, values)
		resultado = cursor.fetchone()
		return resultado
	
	except Error as e:
		flash(f'Error: {e}')
		return None

	finally:
		cursor.close()
		cone.close()
